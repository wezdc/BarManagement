CREATE TABLE IF NOT EXIST VENTAS
(
  id_venta int PRIMARY KEY IDENTITY NOT NULL,
  status bit DEFAULT 0,
  fecha_venta datetime NOT NULL,
--Foreign Keys
  id_venta int NOT NULL,
  id_usuario int NOT NULL,
  id_cliente int NOT NULL,
  
  total_Venta float(2), 
);

CREATE TABLE IF NOT EXIST PRODUCTOS VENTA
(
  id_productos_venta int PRIMARY KEY IDENTITY NOT NULL,
--Foreign Keys  
  id_venta int NOT NULL,
  id_producto int NOT NULL,
  descripcion varchar(20) NOT NULL,
  cantidad int NOT NULL,
  precio float(2) NOT NULL,
  subtotal float(2) NOT NULL
)
